SELECT
    memory_sorts,
    disk_sorts,
    ROUND((memory_sorts / (memory_sorts + disk_sorts)) * 100, 6) AS sort_hit_ratio_percent
FROM (
    SELECT
        SUM(CASE WHEN name = 'sorts (memory)' THEN value ELSE 0 END) AS memory_sorts,
        SUM(CASE WHEN name = 'sorts (disk)' THEN value ELSE 0 END) AS disk_sorts
    FROM v$sysstat
    WHERE name IN ('sorts (memory)', 'sorts (disk)')
);

and 

SELECT event, total_waits, time_waited
FROM v$system_event
WHERE event LIKE '%temp%';
